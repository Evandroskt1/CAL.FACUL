# CAL.FACUL
N1 = int(input('Digite o primeiro numero: '))
N2 = int(input('Digite o segundo numero: '))

#Add 
print("resultado da adicao")
print('{} + {} = '.format(N1, N2))
print(N1 + N2)

 #Sub
print("resultado da subtracao")
print('{} - {} = '.format(N1, N2))
print(N1 - N2)


#Mult
print("resultado da multiplicacao")
print('{} * {} = '.format(N1, N2))
print(N1 * N2)


#Div
print("resultado da divisao")
print('{} / {} = '.format(N1, N2))
print(N1 / N2)
